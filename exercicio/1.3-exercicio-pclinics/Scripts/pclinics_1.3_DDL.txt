CREATE DATABASE VET;
GO
 
USE VET;
GO

CREATE TABLE CLINICA (
idClinica TINYINT PRIMARY KEY IDENTITY (1,1),
endereco VARCHAR (20) NOT NULL,
);
GO

CREATE TABLE VETERINARIO (
idVeterinario SMALLINT PRIMARY KEY IDENTITY (1,1),
idClinica TINYINT FOREIGN KEY REFERENCES CLINICA(idClinica),
);
GO

CREATE TABLE ATENDIMENTO (
idAtendimento SMALLINT PRIMARY KEY IDENTITY (1,1),
idVeterinario SMALLINT FOREIGN KEY REFERENCES VETERINARIO(idVeterinario),
idPets SMALLINT FOREIGN KEY REFERENCES PETS(idPets),
data_inicio DATETIME NOT NULL DEFAULT GETDATE(),
data_fim DATETIME NOT NULL DEFAULT GETDATE(),
);
GO

CREATE TABLE DONO (
idDono TINYINT PRIMARY KEY IDENTITY (1,1),
nome_dono VARCHAR (30) NOT NULL,
);
GO

CREATE TABLE TIPO (
idTipo SMALLINT PRIMARY KEY IDENTITY (1,1),
nome_tipo VARCHAR(30) NOT NULL,
);
GO

CREATE TABLE RACA (
idRaca SMALLINT PRIMARY KEY IDENTITY (1,1),
idTipo SMALLINT FOREIGN KEY REFERENCES TIPO(idTipo),
nome_raca VARCHAR(20) NOT NULL,
);
GO

CREATE TABLE PETS (
idPets SMALLINT PRIMARY KEY IDENTITY (1,1),
idRaca SMALLINT FOREIGN KEY REFERENCES RACA(idRaca),
idDono TINYINT FOREIGN KEY REFERENCES DONO(idDono),
nome_pets VARCHAR(30) NOT NULL,
);
GO
